CREATE DATABASE MuzikPlatformuDB;
GO

USE MuzikPlatformuDB;

CREATE TABLE Ulkeler (
    ulkeID INT IDENTITY(1,1) PRIMARY KEY,
    ulkeAdi NVARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE Cinsiyetler (
    cinsiyetID INT PRIMARY KEY,
    cinsiyetAdi NVARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE Kullanicilar (
    kullaniciID INT IDENTITY(1,1) PRIMARY KEY,
    kullaniciAdi NVARCHAR(50) NOT NULL UNIQUE,
    kullaniciEmail NVARCHAR(100) NOT NULL UNIQUE,
    kullaniciSifreHash NVARCHAR(255) NOT NULL,
    kullaniciDogumTarihi DATE NULL,
    ulkeID INT NULL,
    cinsiyetID INT NULL,
    kayitTarihi DATETIME NOT NULL DEFAULT GETDATE(),
    hesapKapamaTarihi DATETIME NULL,

    CONSTRAINT FK_Kullanicilar_Ulkeler
        FOREIGN KEY (ulkeID) REFERENCES Ulkeler(ulkeID),

    CONSTRAINT FK_Kullanicilar_Cinsiyetler
        FOREIGN KEY (cinsiyetID) REFERENCES Cinsiyetler(cinsiyetID),

    CONSTRAINT CK_Kullanicilar_HesapTarihleri
        CHECK (
            hesapKapamaTarihi IS NULL
            OR hesapKapamaTarihi >= kayitTarihi
        )
);

CREATE TABLE Artistler (
    artistID INT IDENTITY(1,1) PRIMARY KEY,
    artistAdi NVARCHAR(100) NOT NULL,
    artistTipi NVARCHAR(10) NOT NULL,
    baslangicTarihi DATE NULL,
    bitisTarihi DATE NULL,
    ulkeID INT NULL,
    biyografi NVARCHAR(MAX) NULL,
    profilResmiUrl NVARCHAR(500) NULL,
    artistKayitTarihi DATETIME NOT NULL DEFAULT GETDATE(),

    CONSTRAINT FK_Artistler_Ulkeler
        FOREIGN KEY (ulkeID) REFERENCES Ulkeler(ulkeID),

    CONSTRAINT CK_Artistler_ArtistTipi
        CHECK (artistTipi IN (N'Solo', N'Grup')),

    CONSTRAINT CK_Artistler_Tarih
        CHECK (
            bitisTarihi IS NULL
            OR (
                baslangicTarihi IS NOT NULL
                AND bitisTarihi >= baslangicTarihi
            )
        )
);

CREATE TABLE Yayinlar (
    yayinID INT IDENTITY(1,1) PRIMARY KEY,
    yayinAdi NVARCHAR(150) NOT NULL,
    yayinTuru NVARCHAR(20) NOT NULL,
    yayinTarihi DATE NULL,
    kapakResmiUrl NVARCHAR(500) NULL,
    aciklama NVARCHAR(MAX) NULL,
    eklenmeTarihi DATETIME NOT NULL DEFAULT GETDATE(),

    CONSTRAINT CK_Yayinlar_YayinTuru
        CHECK (yayinTuru IN (N'Album', N'Single', N'EP'))
);

CREATE TABLE YayinArtistleri (
    yayinID INT NOT NULL,
    artistID INT NOT NULL,

    PRIMARY KEY (yayinID, artistID),

    CONSTRAINT FK_YayinArtistleri_Yayinlar
        FOREIGN KEY (yayinID) REFERENCES Yayinlar(yayinID),

    CONSTRAINT FK_YayinArtistleri_Artistler
        FOREIGN KEY (artistID) REFERENCES Artistler(artistID)
);

CREATE TABLE Sarkilar (
    sarkiID INT IDENTITY(1,1) PRIMARY KEY,
    sarkiAdi NVARCHAR(150) NOT NULL,
    sarkiSureSaniye INT NOT NULL,
    yayinID INT NULL,
    sarkiDosyaUrl NVARCHAR(500) NULL,
    explicitMi BIT NOT NULL DEFAULT 0,
    eklenmeTarihi DATETIME NOT NULL DEFAULT GETDATE(),

    CONSTRAINT FK_Sarkilar_Yayinlar
        FOREIGN KEY (yayinID) REFERENCES Yayinlar(yayinID),

    CONSTRAINT CK_Sarkilar_Sure
        CHECK (sarkiSureSaniye > 0)
);

CREATE TABLE Etiketler (
    etiketID INT IDENTITY(1,1) PRIMARY KEY,
    etiketAdi NVARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE Playlistler (
    playlistID INT IDENTITY(1,1) PRIMARY KEY,
    kullaniciID INT NOT NULL,
    playlistAdi NVARCHAR(100) NOT NULL,
    aciklama NVARCHAR(500) NULL,
    herkeseAcikMi BIT NOT NULL DEFAULT 1,
    olusturulmaTarihi DATETIME NOT NULL DEFAULT GETDATE(),

    CONSTRAINT FK_Playlistler_Kullanicilar
        FOREIGN KEY (kullaniciID) REFERENCES Kullanicilar(kullaniciID)
);

CREATE TABLE SarkiArtistleri (
    sarkiID INT NOT NULL,
    artistID INT NOT NULL,

    PRIMARY KEY (sarkiID, artistID),

    CONSTRAINT FK_SarkiArtistleri_Sarkilar
        FOREIGN KEY (sarkiID) REFERENCES Sarkilar(sarkiID),

    CONSTRAINT FK_SarkiArtistleri_Artistler
        FOREIGN KEY (artistID) REFERENCES Artistler(artistID)
);

CREATE TABLE SarkiEtiketleri (
    sarkiID INT NOT NULL,
    etiketID INT NOT NULL,

    PRIMARY KEY (sarkiID, etiketID),

    CONSTRAINT FK_SarkiEtiketleri_Sarkilar
        FOREIGN KEY (sarkiID) REFERENCES Sarkilar(sarkiID),

    CONSTRAINT FK_SarkiEtiketleri_Etiketler
        FOREIGN KEY (etiketID) REFERENCES Etiketler(etiketID)
);

CREATE TABLE PlaylistSarkilari (
    playlistID INT NOT NULL,
    sarkiID INT NOT NULL,
    eklenmeTarihi DATETIME NOT NULL DEFAULT GETDATE(),
    siraNo INT NULL,

    PRIMARY KEY (playlistID, sarkiID),

    CONSTRAINT FK_PlaylistSarkilari_Playlistler
        FOREIGN KEY (playlistID) REFERENCES Playlistler(playlistID),

    CONSTRAINT FK_PlaylistSarkilari_Sarkilar
        FOREIGN KEY (sarkiID) REFERENCES Sarkilar(sarkiID),

    CONSTRAINT CK_PlaylistSarkilari_SiraNo
        CHECK (siraNo IS NULL OR siraNo > 0)
);

CREATE TABLE Begeniler (
    kullaniciID INT NOT NULL,
    sarkiID INT NOT NULL,
    begeniTarihi DATETIME NOT NULL DEFAULT GETDATE(),

    PRIMARY KEY (kullaniciID, sarkiID),

    CONSTRAINT FK_Begeniler_Kullanicilar
        FOREIGN KEY (kullaniciID) REFERENCES Kullanicilar(kullaniciID),

    CONSTRAINT FK_Begeniler_Sarkilar
        FOREIGN KEY (sarkiID) REFERENCES Sarkilar(sarkiID)
);

CREATE TABLE DinlemeGecmisi (
    dinlemeID INT IDENTITY(1,1) PRIMARY KEY,
    kullaniciID INT NOT NULL,
    sarkiID INT NOT NULL,
    dinlemeBaslangicZamani DATETIME NOT NULL DEFAULT GETDATE(),
    dinlemeBitisZamani DATETIME NULL,
    dinlenenMilisaniye INT NOT NULL,

    CONSTRAINT FK_DinlemeGecmisi_Kullanicilar
        FOREIGN KEY (kullaniciID) REFERENCES Kullanicilar(kullaniciID),

    CONSTRAINT FK_DinlemeGecmisi_Sarkilar
        FOREIGN KEY (sarkiID) REFERENCES Sarkilar(sarkiID),

    CONSTRAINT CK_DinlemeGecmisi_DinlenenMilisaniye
        CHECK (dinlenenMilisaniye > 0),

    CONSTRAINT CK_DinlemeGecmisi_Zaman
        CHECK (
            dinlemeBitisZamani IS NULL
            OR dinlemeBitisZamani >= dinlemeBaslangicZamani
        )
);

CREATE TABLE SarkiLoglari (
    sarkiLogID INT IDENTITY(1,1) PRIMARY KEY,
    sarkiID INT NOT NULL,
    sarkiAdi NVARCHAR(150) NOT NULL,
    islemTuru NVARCHAR(20) NOT NULL,
    islemTarihi DATETIME NOT NULL DEFAULT GETDATE(),

    CONSTRAINT CK_SarkiLoglari_IslemTuru
        CHECK (islemTuru IN (N'Ekleme', N'Silme'))

GO

CREATE TABLE PlaylistSarkiLoglari (
    playlistSarkiLogID INT IDENTITY(1,1) PRIMARY KEY,
    playlistID INT NOT NULL,
    playlistAdi NVARCHAR(100) NOT NULL,
    sarkiID INT NOT NULL,
    sarkiAdi NVARCHAR(150) NOT NULL,
    islemTuru NVARCHAR(20) NOT NULL,
    islemTarihi DATETIME NOT NULL DEFAULT GETDATE(),

    CONSTRAINT CK_PlaylistSarkiLoglari_IslemTuru
        CHECK (islemTuru IN (N'Ekleme', N'Silme'))
);
GO

CREATE TRIGGER trg_Sarkilar_Insert_Log
ON Sarkilar
AFTER INSERT
AS
BEGIN
    INSERT INTO SarkiLoglari (sarkiID, sarkiAdi, islemTuru)
    SELECT sarkiID, sarkiAdi, N'Ekleme'
    FROM inserted;
END;
GO

CREATE TRIGGER trg_Sarkilar_Delete_Log
ON Sarkilar
AFTER DELETE
AS
BEGIN
    INSERT INTO SarkiLoglari (sarkiID, sarkiAdi, islemTuru)
    SELECT sarkiID, sarkiAdi, N'Silme'
    FROM deleted;
END;
GO

CREATE TRIGGER trg_PlaylistSarkilari_Insert_Log
ON PlaylistSarkilari
AFTER INSERT
AS
BEGIN
    INSERT INTO PlaylistSarkiLoglari
        (playlistID, playlistAdi, sarkiID, sarkiAdi, islemTuru)
    SELECT
        i.playlistID,
        p.playlistAdi,
        i.sarkiID,
        s.sarkiAdi,
        N'Ekleme'
    FROM inserted i
    INNER JOIN Playlistler p ON i.playlistID = p.playlistID
    INNER JOIN Sarkilar s ON i.sarkiID = s.sarkiID;
END;
GO

CREATE TRIGGER trg_PlaylistSarkilari_Delete_Log
ON PlaylistSarkilari
AFTER DELETE
AS
BEGIN
    INSERT INTO PlaylistSarkiLoglari
        (playlistID, playlistAdi, sarkiID, sarkiAdi, islemTuru)
    SELECT
        d.playlistID,
        ISNULL(p.playlistAdi, N'Silinmis Playlist'),
        d.sarkiID,
        ISNULL(s.sarkiAdi, N'Silinmis Sarki'),
        N'Silme'
    FROM deleted d
    LEFT JOIN Playlistler p ON d.playlistID = p.playlistID
    LEFT JOIN Sarkilar s ON d.sarkiID = s.sarkiID;
END;
GO

CREATE VIEW vw_SarkiDetaylari
AS
SELECT
    s.sarkiID,
    s.sarkiAdi,
    s.sarkiSureSaniye,
    s.explicitMi,
    y.yayinAdi,
    y.yayinTuru,
    y.yayinTarihi,
    a.artistAdi,
    a.artistTipi
FROM Sarkilar s
LEFT JOIN Yayinlar y
    ON s.yayinID = y.yayinID
LEFT JOIN SarkiArtistleri sa
    ON s.sarkiID = sa.sarkiID
LEFT JOIN Artistler a
    ON sa.artistID = a.artistID;
GO

CREATE VIEW vw_PlaylistDetaylari
AS
SELECT
    p.playlistID,
    p.playlistAdi,
    k.kullaniciAdi,
    s.sarkiID,
    s.sarkiAdi,
    ps.siraNo,
    ps.eklenmeTarihi
FROM Playlistler p
INNER JOIN Kullanicilar k
    ON p.kullaniciID = k.kullaniciID
INNER JOIN PlaylistSarkilari ps
    ON p.playlistID = ps.playlistID
INNER JOIN Sarkilar s
    ON ps.sarkiID = s.sarkiID;
GO

CREATE VIEW vw_GecerliDinlenmeSayilari
AS
SELECT
    s.sarkiID,
    s.sarkiAdi,
    COUNT(dg.dinlemeID) AS gecerliDinlenmeSayisi
FROM Sarkilar s
LEFT JOIN DinlemeGecmisi dg
    ON s.sarkiID = dg.sarkiID
    AND dg.dinlenenMilisaniye >= 30000
GROUP BY
    s.sarkiID,
    s.sarkiAdi;
GO

CREATE PROCEDURE sp_KullaniciGiris
    @kullaniciEmail NVARCHAR(100),
    @kullaniciSifreHash NVARCHAR(255)
AS
BEGIN
    SELECT
        kullaniciID,
        kullaniciAdi,
        kullaniciEmail
    FROM Kullanicilar
    WHERE kullaniciEmail = @kullaniciEmail
      AND kullaniciSifreHash = @kullaniciSifreHash;
END;
GO

CREATE PROCEDURE sp_PlaylistOlustur
    @kullaniciID INT,
    @playlistAdi NVARCHAR(100),
    @aciklama NVARCHAR(500) = NULL,
    @herkeseAcikMi BIT = 1
AS
BEGIN
    SET NOCOUNT ON;

    IF NOT EXISTS (
        SELECT 1
        FROM Kullanicilar
        WHERE kullaniciID = @kullaniciID
    )
    BEGIN
        RAISERROR(N'Kullanici bulunamadi.', 16, 1);
        RETURN;
    END;

    INSERT INTO Playlistler
        (kullaniciID, playlistAdi, aciklama, herkeseAcikMi)
    VALUES
        (@kullaniciID, @playlistAdi, @aciklama, @herkeseAcikMi);

    SELECT SCOPE_IDENTITY() AS yeniPlaylistID;
END;
GO

CREATE PROCEDURE sp_PlaylisteSarkiEkle
    @playlistID INT,
    @sarkiID INT,
    @siraNo INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    IF NOT EXISTS (
        SELECT 1
        FROM Playlistler
        WHERE playlistID = @playlistID
    )
    BEGIN
        RAISERROR(N'Playlist bulunamadi.', 16, 1);
        RETURN;
    END;

    IF NOT EXISTS (
        SELECT 1
        FROM Sarkilar
        WHERE sarkiID = @sarkiID
    )
    BEGIN
        RAISERROR(N'Sarki bulunamadi.', 16, 1);
        RETURN;
    END;

    IF EXISTS (
        SELECT 1
        FROM PlaylistSarkilari
        WHERE playlistID = @playlistID
          AND sarkiID = @sarkiID
    )
    BEGIN
        RAISERROR(N'Bu sarki zaten playlistte mevcut.', 16, 1);
        RETURN;
    END;

    BEGIN TRANSACTION;

    BEGIN TRY
        IF @siraNo IS NULL
        BEGIN
            SELECT @siraNo = ISNULL(MAX(siraNo), 0) + 1
            FROM PlaylistSarkilari
            WHERE playlistID = @playlistID;
        END;

        INSERT INTO PlaylistSarkilari
            (playlistID, sarkiID, siraNo, eklenmeTarihi)
        VALUES
            (@playlistID, @sarkiID, @siraNo, GETDATE());

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH;
END;
GO

CREATE PROCEDURE sp_SarkiBegen
    @kullaniciID INT,
    @sarkiID INT
AS
BEGIN
    SET NOCOUNT ON;

    IF NOT EXISTS (
        SELECT 1
        FROM Kullanicilar
        WHERE kullaniciID = @kullaniciID
    )
    BEGIN
        RAISERROR(N'Kullanici bulunamadi.', 16, 1);
        RETURN;
    END;

    IF NOT EXISTS (
        SELECT 1
        FROM Sarkilar
        WHERE sarkiID = @sarkiID
    )
    BEGIN
        RAISERROR(N'Sarki bulunamadi.', 16, 1);
        RETURN;
    END;

    IF EXISTS (
        SELECT 1
        FROM Begeniler
        WHERE kullaniciID = @kullaniciID
          AND sarkiID = @sarkiID
    )
    BEGIN
        RAISERROR(N'Bu sarki zaten begenilmis.', 16, 1);
        RETURN;
    END;

    INSERT INTO Begeniler
        (kullaniciID, sarkiID)
    VALUES
        (@kullaniciID, @sarkiID);
END;
GO
